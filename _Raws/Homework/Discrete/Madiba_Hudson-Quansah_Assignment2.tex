\qs{}{
	\[
		\left( p \to q \right) \wedge \left( p \to r \right) \equiv p \to  \left( q \wedge r \right)
	\]
}

\sol{
	\begin{align*}
		\left( p \to q \right) \wedge  \left( p \to r \right) & \equiv \left( \neg p \vee q \right) \wedge \left( \neg p
		\vee r \right) \tag*{Conditional Disjunction Equivalence}                                                        \\
		                                                      & \equiv \neg p \vee \left( q \wedge r \right) \tag*{By
		the First Distributive Law}                                                                                      \\
		                                                      & \equiv p \to \left( q \wedge r \right) \tag*{Conditional
			Conjunction Equivalence}
	\end{align*}
}

\qs{}{
	\[
		p \wedge p \equiv p
	\]
}

\sol{
	\begin{align*}
		p & \equiv p \wedge T \tag*{By the First Identity Law}                                                         \\
		  & \equiv p \wedge \left( p \vee \neg p \right) \tag*{By the First Negation Law}                              \\
		  & \equiv \left( p \wedge p \right) \vee \left( p \wedge \neg p \right) \tag*{By the Second Distributive Law} \\
		  & \equiv \left( p \wedge p \right) \vee F \tag*{By the Second Negation Law}                                  \\
		  & \equiv p \wedge p \tag*{By the Second Identity Law}
	\end{align*}
}
